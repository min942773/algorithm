void upHeap(int * heap, int i){
	int tmp;
	if(i<=1){
		return ;
	}
	if(heap[i]<=heap[i/2]){
		return ;
	}
	else{
		tmp = heap[i];
		heap[i] = heap[i/2];
		heap[i/2] = tmp;
		upHeap(heap, i/2);
	}
}

void downHeap(int * heap, int size, int i){
	int greater_idx, tmp;
	if(i*2>(size)){
		return ;
	}
	greater_idx = i*2;
	if( (i*2+1)<=(size) ){
		if(heap[i*2+1]>heap[greater_idx]){
			greater_idx = i*2+1;
		}
	}
	if(heap[i]>=heap[greater_idx]){
		return ;
	}
	tmp = heap[i];
	heap[i] = heap[greater_idx];
	heap[greater_idx] = tmp;
	downHeap(heap, size, greater_idx);
}

	if(i>(*size)){
		return ;

	int i;
	for(i=(*size)/2;i>0;i--){

void inPlaceHeapSort(int * heap, int * size){
	int i, tmp;
	buildHeap(heap, size);
	for(i=(*size);i>1;i--){
		tmp = heap[1];
		heap[1] = heap[i];
		heap[i] = tmp;
		downHeap(heap, i-1, 1);
	}
}

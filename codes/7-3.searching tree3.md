#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

typedef struct node {
	int key;
	int height;
	struct node * LChild;
	struct node * RChild;
	struct node * Parent;
}node;

node * getNode() {
	node * tmp;
	tmp = (node *)malloc(sizeof(node));
	tmp->LChild = NULL;
	tmp->RChild = NULL;
	tmp->Parent = NULL;
	return tmp;
}

int isExternal(node * myNode) {
	if ((myNode->LChild == NULL) && (myNode->RChild == NULL))
		return 1;
	else
		return 0;
}

void expandExternal(node * z) {
	node * l = getNode();
	node * r = getNode();
	l->Parent = z;
	r->Parent = z;
	l->height = 0;
	r->height = 0;
	z->LChild = l;
	z->RChild = r;
	z->height = 1;
}

node * reduceExternal(node * z, node * root) {
	node * w = getNode();
	node * zs = getNode();
	node * g = getNode();
	w = z->Parent;
	if (z == z->Parent->RChild)
		zs = z->Parent->LChild;
	else
		zs = z->Parent->RChild;
	if (w == root) {
		root = zs;
		zs->Parent = NULL;
	}
	else {
		g = w->Parent;
		zs->Parent = g;
		if (w == g->LChild)
			g->LChild = zs;
		else
			g->RChild = zs;
	}

	return zs;
}

node * inOrderSucc(node * w) {
	node * tmp = getNode();
	tmp = w->RChild;
	while (!isExternal(tmp)) {
		tmp = tmp->LChild;
	}
	tmp = tmp->Parent;
	return tmp;
}

int heightUpdateAndBalanceCheck(node * w) {
	int max;
	int balance;
	node * l = getNode();
	node * r = getNode();
	if (w == NULL)
		return 1;
	l = w->LChild;
	r = w->RChild;
	max = (l->height > r->height) ? l->height : r->height;
	w->height = max + 1;
	balance = (l->height) - (r->height);
	return (balance < 2 && balance > -2);
	/*

}

	else {

	else {
		z->Parent->RChild = b;
		b->Parent = z->Parent;
	}

	a->LChild = tree0;
	tree0->Parent = a;
	a->RChild = tree1;
	tree1->Parent = a;
	max = (tree0->height > tree1->height) ? tree0->height : tree1->height;
	a->height = max + 1;

void searchAndRepairAfterInsertion(node * w, node ** p_root) {
	node * z = getNode();
	node * y = getNode();
	node * x = getNode();
	z = w;
	while (heightUpdateAndBalanceCheck(z)) {
		if (z == *p_root)
			return ;
		z = z->Parent;
	}
	if (z->LChild->height > z->RChild->height)
		y = z->LChild;
	else
		y = z->RChild;
	if (y->LChild->height > y->RChild->height)
		x = y->LChild;
	else
		x = y->RChild;/////////////?????????

void searchAndRepairAfterRemoval(node * w, node ** p_root) {
	node * z = getNode();
	node * y = getNode();
	node * x = getNode();
	node * b = getNode();
	z = w;
	while (heightUpdateAndBalanceCheck(z)) {
		if (z == *p_root) {
			return ;
		}
		z = z->Parent;
	}
	if ((z->LChild->height) > (z->RChild->height))
		y = z->LChild;
	else
		y = z->RChild;

	if ((y->LChild->height) > (y->RChild->height))
		x = y->LChild;
	else if ((y->LChild->height) < (y->RChild->height))
		x = y->RChild;
	else if ((z->LChild) == y)
		x = y->LChild;
	else
		x = y->RChild;
	
	b = restructure(x, y, z, p_root);

	searchAndRepairAfterRemoval(b, p_root);
	return ;
}

node * treeSearch(node * v, int k) {
	if (isExternal(v))
		return v;
	if (k == v->key)
		return v;
	else if (k<v->key)
		return treeSearch(v->LChild, k);
	else
		return treeSearch(v->RChild, k);
}

void findElement(int _key, node ** p_root) {
	node * w = getNode();
	w = treeSearch(*p_root, _key);
	if (isExternal(w))
		printf("X\n");
	else
		printf("%d\n", w->key);
}

void insertItem(node ** p_root, int _k) {
	node * w = getNode();
	w = treeSearch(*p_root, _k);
	if (!isExternal(w))
		return ;
	else {
		w->key = _k;
		expandExternal(w);
		searchAndRepairAfterInsertion(w, p_root);
	}
}

}

void print(node ** p_root) {
	node * pointer = getNode();

	if (isExternal(pointer))
